Criando rotas

Guia para criar rotas

Instale o pacote de rotas do react

 npm install react-router-dom

Inicie uma rota bsica dentro do arquivo index.js

Utilizando links

Use o componente do react Link para redirecionar o usurio:

  <Link to="/">Home</Link>

Criando um gerenciador de rotas

Routing.jsx

import React from "react";
import { Routes, Route, BrowserRouter as Router } from "react-router-dom";
//import Home, About, Topics... etc

const Routing = () => {
	return (
		<Router>
			<Switch>
				<Route path="/"><Home /></Route>
				<Route path="/about"><About /></Route>
		        <Route path="/topics"><Topics /></Route>
			</Switch>
		</Router>
	);
}

export default Routing;

[Snippet] Preenchimento de formulrio

const [user, setUser] = useState({ 
	name: "", email: "", subject: "", message: ""
});

let name, value;

const data = (e) => {
	name = e.target.name;
	value = e.target.value;
	setUser({ ...user, [name]: value });
}

<form method="POST">
	<input
		type="text"
		name="name"
		id="name"
		value={user.name}
		placeholder="Digite seu nome completo"
		required
		autoComplete="off"
		onChange={(e) => data(e)}
	/>
	<input
		type="email"
		name="email"
		id="email"
		value={user.email}
		placeholder="Digite seu endereo de e-mail"
		required
		autoComplete="off"
		onChange={(e) => data(e)}
	/>
	<input
		type="text"
		name="subject"
		id="subject"
		value={user.subject}
		placeholder="Informe o assunto do e-mail"
		required
		autoComplete="off"
		onChange={(e) => data(e)}
	/>
	<textarea
		name="message"
		id="message"
		value={user.message}
		placeholder="Sua mensagem..."
		required
		autoComplete="off"
		onChange={(e) => data(e)}
	>
	</textarea>
</form>
